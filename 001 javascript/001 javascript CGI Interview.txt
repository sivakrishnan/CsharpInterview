################# 02) Remove Duplication in javascript  #################

âœ… 1. Using Set (BEST & MOST USED)

const arr = [1, 2, 2, 3, 4, 4, 5];
const uniqueArr = [...new Set(arr)];
console.log(uniqueArr);


âœ… 2. Using filter() + indexOf()

const arr = [1, 2, 2, 3, 3, 4];
const uniqueArr = arr.filter(
  (value, index) => arr.indexOf(value) === index
);
console.log(uniqueArr);


âœ… 3. Using reduce()

const arr = [1, 1, 2, 3, 3];
const uniqueArr = arr.reduce((acc, curr) => {
  if (!acc.includes(curr)) {
    acc.push(curr);
  }
  return acc;
}, []);

console.log(uniqueArr);


âœ… 4. Remove duplicates from array of objects
ðŸ”¹ Based on a property (id)

const users = [
  { id: 1, name: "A" },
  { id: 2, name: "B" },
  { id: 1, name: "A" }
];

const uniqueUsers = [
  ...new Map(users.map(user => [user.id, user])).values()
];

console.log(uniqueUsers);

âœ… 5. Using for loop (Classic / Interview Safe)

const arr = [1, 2, 2, 3];
const uniqueArr = [];

for (let i = 0; i < arr.length; i++) {
  if (!uniqueArr.includes(arr[i])) {
    uniqueArr.push(arr[i]);
  }
}

console.log(uniqueArr);

################# 04) Spread Operators (expands values) in javascript  #################


âœ… Copy an array (shallow copy)

const arr1 = [1, 2, 3];
const arr2 = [...arr1];
const arr3=arr1;

console.log(arr2); 	// [1, 2, 3]
console.log(arr3);	// [1, 2, 3]

arr3.push(111);
console.log(arr1); //[1, 2, 3, 111]
console.log(arr3===arr1); //true

arr2.push(999);
console.log(arr1); // [1, 2, 3, 111] not added 999

âœ… Merge arrays

const b = [3, 4];

const result = [...a, ...b];
console.log(result);

âœ… Add elements while copying

const nums = [2, 3];
const newNums = [1, ...nums, 4];
console.log(newNums);

âœ… Copy object

âœ… Merge objects

const b = { y: 3, z: 4 };

const merged = { ...a, ...b };
console.log(merged); // { x: 1, y: 3, z: 4 }  â† later overrides earlier

âœ… Add or override properties

const admin = { ...user, role: "Admin" };
console.log(admin);

âœ… Pass array elements as arguments

################# 05) Rest Operators (collects values) in javascript  #################

function sum(a, b, ...rest) {
  console.log(rest); // [3, 4, 5]
  return a + b + rest.reduce((x, y) => x + y, 0);
}

sum(1, 2, 3, 4, 5); // 15

Here:

	a = 1
	b = 2
	rest = [3, 4, 5]
